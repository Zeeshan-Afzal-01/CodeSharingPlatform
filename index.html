JavaScript Fetch API

Reusable fetch wrapper with automatic error handling, response parsing, and request cancellation.

export async function fetchApi(url, options = {}) {
  const controller = new AbortController();
  const { signal } = controller;
  
  const timeoutId = setTimeout(() => controller.abort(), 
    options.timeout || 30000);
    
  try {
    const response = await fetch(url, {
      ...options,
      signal,
      headers: {
        'Content-Type': 'application/json',
        ...options.headers,
      },
    });
    
    clearTimeout(timeoutId);
    
    if (!response.ok) {
      throw new Error(`${response.status}: ${response.statusText}`);
    }
    
    return await response.json();
  } catch (error) {
    // Handle errors
  }
}

Node.js Redis

Implemention of a Redis caching layer for Node.js applications with automatic TTL and serialization.

class RedisCache {
  constructor(redisClient, options = {}) {
    this.client = redisClient;
    this.ttl = options.ttl || 3600; // 1 hour default
    this.prefix = options.prefix || 'cache:';
  }

  async get(key) {
    const data = await this.client.get(`${this.prefix}${key}`);
    if (!data) return null;
    
    try {
      return JSON.parse(data);
    } catch (e) {
      return data;
    }
  }

  async set(key, value, expiry = this.ttl) {
    const data = typeof value === 'object' 
      ? JSON.stringify(value) 
      : value;
      
    return this.client.set(
      `${this.prefix}${key}`, 
      data, 
      'EX', 
      expiry
    );
  }
}

Migration script to transform user data schema from v1 to v2, including data validation and rollback mechanism.

import psycopg2
from psycopg2.extras import RealDictCursor
import logging

# Configure logging
logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    filename='migration.log'
)

def migrate_users_schema():
    """Migrates users from old schema to new schema"""
    
    conn = None
    try:
        # Connect to the PostgreSQL database
        conn = psycopg2.connect(
            host="localhost",
            database="app_db",
            user="postgres",
            password="password"
        )
        
        # Create a cursor
        cur = conn.cursor(cursor_factory=RealDictCursor)
        
        # Start a transaction
        conn.autocommit = False
        
        # Step 1: Create backup table
        logging.info("Creating backup table...")
        cur.execute("""
            CREATE TABLE users_backup AS 
            SELECT * FROM users;
        """)
